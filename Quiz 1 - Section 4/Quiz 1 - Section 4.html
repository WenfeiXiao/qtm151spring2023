# Write your own code
list_names =["data_quiz/wdi_co2_2005.csv", "data_quiz/wdi_co2_2010.csv", "data_quiz/wdi_co2_2015.csv"]
list_years = ["2005", "2010", "2015"]
index =0 
for datasetname in list_names:
    dataset = pd.read_csv( datasetname)
    co2_gasfuel_kt = dataset["co2_gasfuel_kt"]
    co2_liquidfuel_kt = dataset["co2_liquidfuel_kt"]
    co2_total_kt = co2_gasfuel_kt+co2_liquidfuel_kt
    plt.hist(x= co2_total_kt)
    plt.ylabel("Frequency")
    plt.xlabel("co2_total_kt")
    plt.title("Dataset " +  list_years[index]  + " total CO2 emissions driven by gas and liquid fuel")
    plt.show()
    index = index + 1
